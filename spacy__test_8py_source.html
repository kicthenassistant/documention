42def get_unit(sentence, lista):
43 """
44 @brief Extrai o ingrediente de um frase.
45 @details Identifica um ingrediente numa frase dada uma lista de possíveis ingredientes.
46
47 @param sentence A sentença a ser analisada.
48 @param lista Lista de ingredientes conhecidos.
49
50 @return O ingrediente identificado na sentença; None se nenhum correspondente for encontrado.
51 """
52 # Process the sentence using spaCy to tokenize and lemmatize the text
53 doc = nlp(sentence.lower())
54
55 # Convert list to a set of lemmas and original text to improve matching chances
56 lemmatized_list = {nlp(word.lower())[0].lemma_ for word in lista}
57 original_list = set(lista)
58
59 # Check for multi-word units first
60 for unit in [unit for unit in lista if ' ' in unit]:
61 if unit.lower() in sentence.lower():
62 return unit
63
64 # Check for single-word units using lemmatization and direct match
65 for token in doc:
66 if token.lemma_ in lemmatized_list or token.text in original_list:
67 return token.text # return the original text from the doc
68
69 return None # If no unit is found
