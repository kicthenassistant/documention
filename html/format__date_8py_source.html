48 @details Esta função utiliza expressões regulares para identificar os formatos de datas suportados e extrair os componentes da data.
49 Em seguida, cria um objeto datetime com a data fornecida e a converte para o formato ISO 8601.
50
51 @param date_string <string> contendo a data num dos formatos suportados.
52
53 @return <string> com a data no formato ISO ou uma mensagem indicando formato inválido.
54
55 @note Caso a string de data não corresponda a nenhum dos formatos predefinidos, a função retornará "Invalid date format".
56 """
57 for pattern, date_format in date_patterns.items():
58 match = re.match(pattern, date_string.strip())
59 if match:
60 if date_format == "DD/MM/YYYY" or date_format == "DD/MM YYYY":
61 day, month, year = match.groups()
62 return datetime(int(year), int(month), int(day)).strftime('%Y-%m-%d')
63 elif date_format == "DD de Month de YYYY":
64 day, month_name, year = match.groups()
65 month = month_to_number.get(month_name.capitalize(), None)
66 if month:
67 return datetime(int(year), month, int(day)).strftime('%Y-%m-%d')
68 return "Invalid date format"
