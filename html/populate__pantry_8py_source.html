120def populate_pantry():
121 """
122 @brief Função para povoar a base de dados pantry_db.
123 @details Insere produtos na base de dados com quantidades e datas de validade aleatórias.
124 Cada produto pode ter entre 0 e 1 entradas com quantidades de 1 a 4 e datas de validade
125 que variam de 30 a 730 dias a partir da data atual.
126
127 @note Esta função utiliza a função `insertStock` do módulo `pantrydb_queries` para inserir os produtos na base de dados.
128 """
129 today = datetime.now()
130 for name, unit in products:
131 # Randomize the quantity and create between 1 and 3 entries per product
132 for _ in range(random.randint(0, 1)):
133 quantity = random.randint(1, 4) # Random quantity between 1 and 4
134 # Generate a random expiration date within the next two years
135 expiration_date = today + timedelta(days=random.randint(30, 730))
136 expiration_date_str = expiration_date.strftime('%Y-%m-%d')
137 # Insert the stock item into the database
138 pdb.insertStock(name, quantity, unit, expiration_date_str)
