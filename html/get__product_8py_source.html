127def get_ingredient(sentence):
128 """
129 @brief Processa uma frase para identificar e retornar o ingrediente mencionado.
130 @details Esta função utiliza o modelo de processamento de linguagem natural do spaCy para lematizar e analisar a sentença, identificando o ingrediente mencionado com base em uma lista predefinida de alimentos.
131 A função verifica se a sentença contém ingredientes completos ou suas formas lematizadas, retornando o ingrediente correspondente.
132
133 @param sentence <string> contendo a sentença a ser analisada.
134
135 @return <string> com o ingrediente identificado ou None se nenhum ingrediente for encontrado.
136
137 @note Caso não seja encontrado um ingrediente completo, a função tentará identificar ingredientes flexionados ou lematizados.
138 Se mesmo assim nenhum ingrediente for encontrado, a função retornará None.
139 """
140
141 # Process the sentence using spaCy to tokenize and lemmatize the text
142 doc = nlp(sentence.lower())
143
144 # Convert list to lemmas
145 lemmatized_list = {nlp(word)[0].lemma_ for word in food_list}
146 original_list = set(food_list)
147
148 # Check for multi-word units first (assumes lista is already lemmatized if necessary)
149 for unit in [unit for unit in food_list if ' ' in unit]:
150 if unit in sentence:
151 return unit
152
153 # Check for single-word units using lemmatization
154 for token in doc:
155 if token.lemma_ in lemmatized_list or token.text in original_list:
156 return token.text # return the original text
157
158 return None # If no unit is found

160def get_unit(sentence):
161 """
162 brief Processa uma frase para identificar e retornar a unidade de medida mencionada, se houver.
163 @details Esta função verifica se a sentença contém uma unidade de medida comum, retornando a unidade identificada.

165
166 @param sentence <string> contendo a sentença a ser analisada.
167
168 @return <string> com a unidade de medida identificada ou None se nenhuma unidade for encontrada.
169
170 @note Caso não seja encontrada uma unidade de medida composta, a função tentará identificar unidades de medida simples.
171 """
172 words = sentence.split() # --------------------------------------- Split the sentence into words
