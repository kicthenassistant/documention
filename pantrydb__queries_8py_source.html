102# Convert a quantity from one unit to another using a dictionary of conversion factors
103def convert_measure(quantity, from_unit, to_unit, conversion_factors):
104 """
105 @brief Convert Uma quantidade de uma unidade para outra.
106 @details Converte uma quantidade de uma unidade para outra, usando um dicionário de fatores de conversão.
107 Trata casos diretos e requer conversões recursivas se necessário.
108
109 @param quantity <int> Quantidade a ser convertida.
110 @param from_unit <string> Unidade inicial.
111 @param to_unit <string> Unidade final.
112 @param conversion_factors <dict> Dicionário contendo os fatores de conversão.
113
114 @return <tuplo> contendo a quantidade convertida e a unidade final.
115
116 @note Funcção usada para podermos subtrair/adicionar quantidades em unidades diferentes do mesmo produto
117
118 @warning ValueError Se não encontrar um caminho de conversão válido.
119 """
120 from_unit = from_unit.lower().strip('s') # Normalize the from_unit name
121 to_unit = to_unit.lower().strip('s') # Normalize the to_unit name
122
123 def find_conversion(current_quantity, current_unit, target_unit, visited_units=set()):
124 # Direct conversion case
125 if target_unit in conversion_factors[current_unit]:
126 return current_quantity * conversion_factors[current_unit][target_unit], target_unit
127
128 # Recursive conversion case: find a path through an intermediary
129 for intermediate_unit in conversion_factors[current_unit]:
130 if intermediate_unit not in visited_units: # Avoid cycles
131 visited_units.add(intermediate_unit)
132 intermediate_quantity = current_quantity * conversion_factors[current_unit][intermediate_unit]
133 # Recursively convert from intermediate unit to target unit
134 result = find_conversion(intermediate_quantity, intermediate_unit, target_unit, visited_units)
135 if result:
136 return result
137 return None
138
139 # Start the conversion process
140 result = find_conversion(quantity, from_unit, to_unit)
141 if result:
142 return result
143 else:
144 raise ValueError("No valid conversion path found from {} to {}".format(from_unit, to_unit))
